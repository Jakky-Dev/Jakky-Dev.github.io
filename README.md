    function generateManifest() {
        const inputBox = document.getElementById('fileInput');
        const button = document.getElementById('submitBtn');
        let gameId = inputBox.value.trim();
    
        const logBox = document.getElementById('logBox');
        logBox.innerHTML = ""; 
        createLogEntry("Generating...", "info");
        button.disabled = true;
    
        if (!gameId) {
            createLogEntry("Please enter a game ID or URL.", "error");
            button.disabled = false;
            return;
        }
    
        const urlPattern = /https:\/\/store\.steampowered\.com\/agecheck\/app\/(\d+)\//;
        const match = gameId.match(urlPattern);
        if (match) {
            gameId = match[1];
        }
    
        if (!/^\d+$/.test(gameId)) {
            createLogEntry("Invalid ID. Please enter a valid numerical game ID.", "error");
            button.disabled = false;
            return;
        }
    
        setTimeout(() => {
            const url = `https://lightcloud.click/api/gameid/${gameId}.zip`;
            fetch(url)
                .then(response => {
                    if (response.ok) {
                        createLogEntry("Success! Manifest generated.", "success");
                        return response.blob();
                    } else {
                        createLogEntry("This game is not available in the database.", "error");
                        throw new Error("Game not found");
                    }
                })
                .then(blob => {
                    const a = document.createElement('a');
                    const downloadUrl = URL.createObjectURL(blob);
                    a.href = downloadUrl;
                    a.download = `${gameId}.zip`;
                    a.click();
                    URL.revokeObjectURL(downloadUrl);
                })
                .catch(error => {
                    createLogEntry("Failed to generate manifest.", "error");
                    console.error('Error:', error);
                })
                .finally(() => {
                    button.disabled = false;
                });
        }, 3000);
